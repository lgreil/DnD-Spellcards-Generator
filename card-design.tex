% =========================
% SPELL COMMAND
% =========================
% Arguments:
% 1 = Spell name
% 2 = Spell school/color
% 3 = Level & school text
% 4 = Description (itemize allowed)
% 5 = Higher-level text (optional)
% 6 = Optional tcolorbox options

\newcommand{\spell}[6]{
\begin{spellcard}[#6]
{\titlefont\color{#2}\textbf{#1}}\\

\vspace{2mm}
\hrule
\vspace{1mm}

\small
\actionicon{} 1 Aktion \hfill \rangeicon{} Selbst (4,5 m Kegel)\\
\saveicon{} GE-RW \hfill \durationicon{} Sofort\\
\componenticon{} V, G

\vspace{1mm}
\hrule
\vspace{1mm}

#4

\ifx #5\empty
% do nothing if #5 is empty
\else
\vspace{1mm}
\hrule
\vspace{1mm}
\small #5
\fi

\end{spellcard}
}
