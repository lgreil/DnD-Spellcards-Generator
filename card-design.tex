\newcommand{\parsecomponents}[1]{%
  \begingroup\footnotesize\color{gray}%
  \IfSubStr{#1}{V}{\faVolumeUp\kern0.5mm}{}%
  \IfSubStr{#1}{S}{\faHandPaper\kern0.5mm}{}%
  \IfSubStr{#1}{G}{\faHandPaper\kern0.5mm}{}%
  \IfSubStr{#1}{M}{\faGem\kern0.5mm}{}%
  \textbf{(\StrDel{#1}{ })}\endgroup}

% ------------------------------------------------------------------
% HEADER WITH ICON (top-right aligned)
% ------------------------------------------------------------------
\newcommand{\spellheader}[3]{%
  % #1 = title
    % #2 = school key (color + icon)
    % #3 = subtitle (level etc.)

    \noindent
    \begin{minipage}{0.78\linewidth}
      {\titlefont\small\color{#2}\textbf{#1}}\\
      {\tiny\color{gray}#3}
    \end{minipage}%
    \begin{minipage}{0.2\linewidth}
      \raggedleft
      {\Large\color{#2}\spellschoolicon{#2}}
    \end{minipage}

    \vspace{1.5mm}
    {\color{#2}\hrule}
}

  }

    \vspace{1.5mm}
}
