% 3. SPELL HEADER (uses unified color + icon)
% ------------------------------------------------------------------
\newcommand{\spellheader}[3]{%
  % #1 = title
  % #2 = school key
  % #3 = subtitle (level etc.)

  \noindent
  \begin{minipage}{0.78\linewidth}
    {\titlefont\small\color{\schoolcolor{#2}}\textbf{#1}}\\
    {\tiny\color{gray}#3}
  \end{minipage}%
  \begin{minipage}{0.2\linewidth}
    \raggedleft
    {\Large\color{\schoolcolor{#2}}\schoolicon{#2}}
  \end{minipage}

  \vspace{1.5mm}
  {\color{\schoolcolor{#2}}\hrule}
}
% #1 = Action
% #2 = Range
% #3 = Components (V,S,G,M)
% #4 = Duration
% #5 = Optional Material (can be empty)
\newcommand{\spellmeta}[5]{%
  {\footnotesize
    % Top row: Action and Range
    \actionicon\ #1 \hfill \rangeicon\ #2
    \par
    % Second row: Components icons + letters, and Duration
    \parsecomponents{#3} \hfill \durationicon\ #4
    \par
    % Third row: Material in italics, only if provided
    \ifx&#5&%
      % nothing
    \else
      {\footnotesize\itshape #5\par}%
    \fi
  }%
  \vspace{1.5mm}
  {\color{gray}\hrule}
  \vspace{1.5mm}
}
